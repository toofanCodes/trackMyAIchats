ðŸ”§ Stored Procedure

sql
DELIMITER //

CREATE PROCEDURE ProcessEmployeeBonus(
    IN in_emp_id INT,
    OUT out_status VARCHAR(100)
)
BEGIN
    DECLARE emp_salary DECIMAL(10,2);
    DECLARE bonus DECIMAL(10,2);
    DECLARE EXIT HANDLER FOR SQLEXCEPTION
    BEGIN
        ROLLBACK;
        SET out_status = 'Error: Transaction failed';
    END;

    START TRANSACTION;

    -- Check if employee exists and get salary
    SELECT salary INTO emp_salary
    FROM employees
    WHERE emp_id = in_emp_id
    FOR UPDATE;

    IF emp_salary IS NULL THEN
        ROLLBACK;
        SET out_status = 'Error: Employee not found';
    ELSE
        -- Calculate bonus (10% of salary)
        SET bonus = emp_salary * 0.10;

        -- Insert into bonuses table
        INSERT INTO bonuses(emp_id, bonus_amount, created_at)
        VALUES (in_emp_id, bonus, NOW());

        -- Mark employee as having received bonus
        UPDATE employees
        SET bonus_paid = TRUE
        WHERE emp_id = in_emp_id;

        COMMIT;
        SET out_status = CONCAT('Success: Bonus of $', bonus, ' processed');
    END IF;
END //

DELIMITER ;
